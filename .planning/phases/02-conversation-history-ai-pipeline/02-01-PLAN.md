```typescript
export async function runComplianceFlow(contactId: string): Promise<boolean> {
  const log = logger.child({ contactId, service: 'compliance' });
  const session = getOrCreateSession(contactId);

  if (!session.disclosureSent) {
    log.info('new contact — sending AI disclosure and LGPD consent prompt');
    await sendMessage(contactId, env.DISCLOSURE_MESSAGE);
    await sendMessage(contactId, env.LGPD_CONSENT_MESSAGE);
    session.disclosureSent = true;
    // consentGiven stays false — the user's NEXT message is the implicit consent (D-05/D-06).
    return false;
  }

  if (!session.consentGiven) {
    // D-05/D-06: any subsequent message = implicit consent. Proceed to AI.
    session.consentGiven = true;
    log.info('implicit LGPD consent recorded (any reply after consent prompt)');
    return true;
  }

  log.debug('compliance already satisfied for contact; proceeding');
  return true;
}
```

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Information disclosure | sessionService.ts logs | mitigate | sessionService must NOT log contents of history or consent state; this plan's implementation imports no logger — state mutations logged only by callers (complianceService uses logger.child({contactId}) without dumping session fields) |
| T-02-02 | Tampering | SessionState mutation | accept | Single-process, in-memory Map; no external mutation vector. Plan 02 adds mutex to prevent interleaved writes. This plan's functions are synchronous — no await between Map.get and Map.set in getOrCreateSession, so race conditions impossible within this file |
| T-02-03 | Repudiation | Compliance state reset via __resetComplianceStoreForTesting | mitigate | The shim calls __resetSessionStoreForTesting; JSDoc labels both "Test-only helper; do not call in production"; no runtime gate added (solo developer workflow — reviewed at code review time; not a public HTTP surface) |
